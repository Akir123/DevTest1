public class PrimeNumber {

    public static void main(String[] args) {
        int n = 50; //check number (check kar)
        boolean isPrime = isPrime(n, 2);
        
        if (isPrime) {
            System.out.println(n + " is a prime number.");
        } else {
            System.out.println(n + " is not a prime number.");
        }
    }

    // Recursive method to check if a number is prime
    static boolean isPrime(int n, int i) {
        // Base case 1: If n is less than or equal to 1, it's not prime
        if (n <= 1) {
            return false;
        }
        // Base case 2: If n is less than or equal to 2, it is prime
        if (n == 2) {
            return true;
        }
        // If n is divisible by i, it's not prime
        if (n % i == 0) {
            return false;
        }
        // Base case 3: If i*i is greater than n, n is prime
        if (i * i > n) {
            return true;
        }
        // Recursive case: Check next possible divisor
        return isPrime(n, i + 1);
    }
}
